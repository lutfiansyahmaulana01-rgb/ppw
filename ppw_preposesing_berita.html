import pandas as pd
import numpy as np
from gensim.models import Word2Vec
from tqdm import tqdm
import ast

# === 1. Baca data ===
df = pd.read_excel("data_berita_tokenized.xlsx")

# Pastikan kolom yang tersedia
print("Kolom tersedia:", df.columns.tolist())

# Tentukan nama kolom label
label_col = "label"  # pastikan sesuai nama di Excel

# === 2. Pastikan kolom 'tokenized' sudah berbentuk list ===
def safe_eval(x):
    """Ubah string list ke list Python, atau biarkan jika sudah list."""
    if isinstance(x, list):
        return x
    if isinstance(x, str):
        try:
            val = ast.literal_eval(x)
            if isinstance(val, list):
                return val
            else:
                return x.split()  # fallback jika ternyata string biasa
        except Exception:
            # kalau gagal eval, anggap string biasa dipisah spasi
            return x.split()
    return []

df['tokenized'] = df['tokenized'].apply(safe_eval)

# === 3. Siapkan data untuk training CBOW ===
sentences = df['tokenized'].tolist()

# === 4. Train Word2Vec model (CBOW) ===
print("Melatih model Word2Vec (CBOW)...")
model = Word2Vec(
    sentences=sentences,
    vector_size=100,  # ukuran dimensi vektor
    window=5,         # konteks kata
    min_count=1,      # minimal kemunculan kata
    workers=4,
    sg=0              # 0 = CBOW, 1 = Skip-Gram
)

# === 5. Konversi setiap dokumen menjadi vektor (rata-rata kata) ===
def get_avg_vector(tokens, model):
    """Ambil rata-rata vektor dari token dalam satu dokumen."""
    vectors = [model.wv[word] for word in tokens if word in model.wv]
    if not vectors:
        return np.zeros(model.vector_size)
    return np.mean(vectors, axis=0)

# Terapkan dengan tqdm agar ada progress bar
tqdm.pandas(desc="Membuat representasi dokumen")
doc_vectors = df['tokenized'].progress_apply(lambda tokens: get_avg_vector(tokens, model))
doc_vectors = np.vstack(doc_vectors.values)

# === 6. Simpan hasil ke Excel dengan label di kolom terakhir ===
cbow_df = pd.DataFrame(doc_vectors)
cbow_df[label_col] = df[label_col].values

# Pastikan kolom label di paling akhir
cols = [c for c in cbow_df.columns if c != label_col] + [label_col]
cbow_df = cbow_df[cols]

cbow_df.to_excel("news_cbow_with_label.xlsx", index=False)
print("✅ Hasil CBOW disimpan ke 'news_cbow_with_label.xlsx'")

import pandas as pd
import numpy as np
from gensim import corpora
from gensim.models import LdaModel

# === 1. Baca data mentah ===
df = pd.read_excel("data_berita_tokenized.xlsx")

if "article" not in df.columns or "label" not in df.columns:
    raise ValueError("Kolom 'article' atau 'label' tidak ditemukan!")

# === 2. Tokenisasi ===
texts = df["article"].astype(str).apply(lambda x: x.split()).tolist()

# === 3. Buat kamus & corpus ===
dictionary = corpora.Dictionary(texts)
dictionary.filter_extremes(no_below=2, no_above=0.8)
corpus = [dictionary.doc2bow(text) for text in texts]

# === 4. Latih model LDA ===
num_topics = 20
lda_model = LdaModel(
    corpus=corpus,
    id2word=dictionary,
    num_topics=num_topics,
    random_state=42,
    passes=10,
    iterations=100,
)

# === 5. Konversi ke vektor topik ===
def get_topic_vector(lda_model, bow, num_topics):
    topic_probs = [0.0] * num_topics
    for topic_id, prob in lda_model.get_document_topics(bow):
        topic_probs[topic_id] = prob
    return np.array(topic_probs)

X_lda = np.array([get_topic_vector(lda_model, bow, num_topics) for bow in corpus])

# === 6. Simpan hasil ke CSV ===
df_lda = pd.DataFrame(X_lda, columns=[f"Topic_{i+1}" for i in range(num_topics)])
df_lda["label"] = df["label"].values
df_lda.to_excel("news_lda.xlsx", index=False)

print("✅ Hasil vektor LDA disimpan ke 'news_lda.xlsx'")
