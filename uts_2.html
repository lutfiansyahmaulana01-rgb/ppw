import pandas as pd
from sklearn.feature_extraction.text import TfidfVectorizer

# === 1. Baca hasil tokenisasi ===
df = pd.read_csv("spam_tokenizing.csv")

# Pastikan kolomnya ada
if "Text" not in df.columns:
    raise ValueError("Kolom 'Text' tidak ditemukan di dataset!")

# === 2. Bersihkan data: ubah ke string dulu ===
df["Text"] = df["Text"].astype(str)

# === 3. TF-IDF Vectorization ===
vectorizer = TfidfVectorizer(
    max_features=10000,        # jumlah fitur maksimal (bisa ubah sesuai kebutuhan)
    stop_words='english',   # hapus stopword bahasa Indonesia
)

X = vectorizer.fit_transform(df["Text"])

# === 4. Simpan hasil TF-IDF ke DataFrame ===
tfidf_df = pd.DataFrame(X.toarray(), columns=vectorizer.get_feature_names_out())

# Jika ada kolom kategori, ikut gabungkan
if "Label" in df.columns:
    tfidf_df["Label"] = df["Label"]

# === 5. Simpan hasil ke CSV ===
tfidf_df.to_csv("spam_tfidf.csv", index=False)

print(" File 'spam_tfidf.csv' berhasil dibuat dengan fitur TF-IDF.")

import pandas as pd
from sklearn.cluster import KMeans
from sklearn.decomposition import PCA
import matplotlib.pyplot as plt
import numpy as np

# === 1. Read the TF-IDF file ===
df_tfidf = pd.read_csv("spam_tfidf.csv")

# Separate features and label if it exists
if "Label" in df_tfidf.columns:
    X = df_tfidf.drop(columns=["Label"])
else:
    X = df_tfidf

# === 2. K-Means clustering ===
n_clusters = 2  # assuming Spam vs Non-Spam
kmeans = KMeans(n_clusters=n_clusters, random_state=42)
df_tfidf["Cluster"] = kmeans.fit_predict(X)

# === 3. Find top words for each cluster ===
feature_names = X.columns
cluster_centers = kmeans.cluster_centers_

topn = 10  # top N words per cluster
for i in range(n_clusters):
    print(f"\n Cluster {i}:")
    top_indices = cluster_centers[i].argsort()[-topn:][::-1]
    top_words = [feature_names[j] for j in top_indices]
    print("Top words:", ", ".join(top_words))

# === 4. Optional: visualize clusters using PCA ===
pca = PCA(n_components=2)
reduced = pca.fit_transform(X)
plt.figure(figsize=(8,6))
plt.scatter(reduced[:,0], reduced[:,1], c=df_tfidf["Cluster"], cmap="coolwarm")
plt.title("K-Means Clustering of Spam vs Non-Spam (TF-IDF)")
plt.xlabel("PCA 1")
plt.ylabel("PCA 2")
plt.show()

# === 5. Save the clustered file ===
df_tfidf.to_csv("spam_kmeans.csv", index=False)
print("\n File 'spam_kmeans.csv' saved with cluster labels.")
